def remove_outlier(df, cutoff = 0.01):
    q_low = df.quantile(cutoff)
    q_hi = df.quantile(1 - cutoff)
    
    df_trim = df.loc[(df > q_low) & (df < q_hi)]
    
    return(df_trim)
